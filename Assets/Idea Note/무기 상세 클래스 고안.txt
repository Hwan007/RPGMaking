        public enum MeleeType
        {
            Axe,
            Sword,
            Spear
        }

        public class MeleeWeaponPart
        {
            public float Length;
            public int MaterialLevel;
        }
        public WeaponType Type = WeaponType.Melee;
        public MeleeType AttackPart;
        public MeleeWeaponPart Handle;
        public MeleeWeaponPart Blade;
        public override string GetDescription()
        {
            string desc = base.GetDescription();

            desc += "\n";
            desc += $"Type : {Type} - {AttackPart}\n";
            desc += $"Handle\n";
            desc += $"   Length : {Handle.Length}\n";
            desc += $"   Material Level : {Handle.MaterialLevel}\n";
            desc += $"Blade\n";
            desc += $"   Length : {Blade.Length}m\n";
            desc += $"   Material Level : {Blade.MaterialLevel}m\n";

            return desc;
        }
